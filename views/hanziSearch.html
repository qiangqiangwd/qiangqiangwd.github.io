汉字模糊搜索

在不进行上传服务器的情况下，在前端进行搜索

当在数据全部都加载出来的情况下，进行汉字的搜索（默认可以也可以使用拼音进行搜索），这里要注意的当是 拼音搜索时，函数会将所有的内容都转换成小写（包括输入的内容）的拼音字母进行判断，然后返回搜索的内容。 例如当输入的搜索内容为'han',下面出现的会有'张三(zhangshan)', '憨子(hanzi)'

内容：

    (function(){
        var searchFun = function(cnt,allArr){
            return new searchFun.prototype.init(cnt,allArr)
        };

        searchFun.fn = searchFun.prototype = {
            init:function(cnt,allArr){
                if(!cnt || cnt == '') return allArr; //若搜索内容为空或则错误时，返回所有内容
                var opts = [],//返回的搜寻的结果
                     isChn = this.isChinese(cnt);  //先对要搜索的内容进行判断（是汉字还是拼音）

                for(var i = 0;i < allArr.length;i++){
                    if(isChn){
                        //参数是汉字
                        if(allArr[i].indexOf(cnt) != -1){
                            //将匹配到的参数进行赋值
                            opts.push(allArr[i]);
                        }
                    }else{
                        //参数是拼音（会将字母转化为小写）时，将所有的数据转换成拼音（小写）的形式，
                        if(this.ConvertPinyin(allArr[i]).indexOf(cnt.toLowerCase()) != -1){
                            //将匹配到的参数进行赋值
                            opts.push(allArr[i]);
                        }
                    }
                }
                return opts;
            },
            //是返回 true，不是返回 false
            isChinese:function(str){
                var cnChar=str.match(/[^\x00-\x80]/g);
                return !!(cnChar!=null&&cnChar.length>0);
            },
            //汉字转拼音
            ConvertPinyin:function(l1){
                var l2 = l1.length;
                var I1 = "";
                var reg = new RegExp('[a-zA-Z0-9\- ]');
                for (var i = 0; i < l2; i++) {
                    var val = l1.substr(i, 1);
                    var name = this.arraySearch(val, PinYin);
                    if (reg.test(val)) {
                        I1 += val;
                    } else if (name !== false) {
                        I1 += name;
                    }
                }
                I1 = I1.replace(/ /g, '-');
                while (I1.indexOf('--') > 0) {
                    I1 = I1.replace('--', '-');
                }
                return I1;
            }
            //在这里参数 PinYin 是引用的另外的参数
            // 在对象中搜索,将中文转化为拼音
            ,arraySearch:function(l1){
                for (var name in PinYin) {
                    if (PinYin[name].indexOf(l1) != -1) {
                        return name;
                        break;
                    }
                }
                return false;
            },
            //首字母大写
            ucfirst:function(l1){
                if (l1.length > 0) {
                    var first = l1.substr(0, 1).toUpperCase();
                    var spare = l1.substr(1, l1.length);
                    return first + spare;
                }
            }
        };

        searchFun.fn.init.prototype = searchFun.fn;
        window.searchFun = searchFun;
    })();
